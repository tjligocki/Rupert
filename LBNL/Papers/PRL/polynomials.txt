Let k = l^2

m = 1:

n mod 1 = 0 (all n):
(1,n):
    p(k) = k - n
    k = n


m = 2:

n mod 2 = 0 (even n), n = 2 j, (2,n) -> (1,n/2) -> (1,j): 
  p(k) = k - n/2 = k - j
  k = n/2 = j

n mod 2 = 1 (odd n), n = 2 j + 1
(2,n):
  p(k) = k - n/2 + 3/8 = k - j - 1/8
  k = n/2 - 3/8 = j + 1/8


m = 3:

n mod 3 = 0, n = 3 j, (3,n) -> (1,n/3) -> (1,j):
  p(k) = k - n/3 = k - j
  k = n/3 = j

n mod 3 = 1, n = 3 j + 1, n = 4, 7:
  see below

n mod 3 = 1, n = 3 j + 1, n >= 10:
  p(k) = k^2 - (2 (n + 44) / 3) k + (n^2 + 88 n - 80) / 9 = k^2 - (2 (j + 15)) k + (j^2 + 30 j + 1)
  k = n/3 - (12 sqrt(14) - 44) / 3 = j - (4 sqrt(14) - 15)

  Examples:

    p(10) = k^2 - 36 k + 100, k = 18 - 4 sqrt(14)
    p(13) = k^2 - 38 k + 137, k = 19 - 4 sqrt(14)
    p(16) = k^2 - 40 k + 176, k = 20 - 4 sqrt(14)

n mod 3 = 2, n = 3 j + 2:
  p(k) = k^2 - (2 (n + 28) / 3) k + (n^2 + 56 n - 80) / 9 = k^2 - (2 (j + 10)) k + (j^2 + 20 j + 4)
  k = n/3 - (12 sqrt(6) - 28) / 3 = j - (4 sqrt(6) - 10)

  Examples:
    p(5)  = k^2 - 22 k + 25, k = 11 - 4 sqrt(6)
    p(8)  = k^2 - 24 k + 48, k = 12 - 4 sqrt(6)
    p(11) = k^2 - 26 k + 73, k = 13 - 4 sqrt(6)


m = 4:

n mod 4 = 0, n = 4 j, (4,n) -> (1,n/4) -> (1,j):
  p(k) = k - n/4 = k - j
  k = n/4 = j

n mod 4 = 1, n = 4 j + 1:
  n = 5  -> Have numerical value for k
  n = 9  -> Have numerical value for k
  n = 13 -> Almost have numerical value for k

n mod 4 = 2, n = 4 j + 2, (4,n) -> (2,n/2) -> (2,2j+1):
  p(k) = k - n/4 + 3/8 = k - j - 1/8
  k = n/4 - 3/8 = j + 1/8

n mod 4 = 3, n = 4 j + 3:
  p(k) = k^2 - (2 (n + 19) / 4) k + (n^2 + 38 n + 73) / 16 = k^2 - (11 + 2j) k + (j^2 + 11 j + 49/4)
  k = n/4 - (12 sqrt(2) - 19)/4 = j - (3 sqrt(2) - 11/2)

  Examples:
    p(7)  = k^2 - 13 k +  97/4, k = 11/2 - 3 sqrt(2)
    p(11) = k^2 - 15 k + 153/4, k = 13/2 - 3 sqrt(2)
    p(15) = k^2 - 17 k + 217/4, k = 15/2 - 3 sqrt(2)


All m:

n mod m = 0, n = m j, (m,n) -> (1,n/m) -> k - n/m = 0 ->
  p(k) = k - n/m = k - j
